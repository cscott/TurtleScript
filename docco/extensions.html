    var object, result;
    if (typeof(this.prototype)==="object") {
        object = Object.create(this.prototype);
    } else {
        object = {};
    }
    result = this.apply(object, arguments);
    if (typeof(result)==="object") {
        return result;
    }
    return object;
};
